# SEO

使用 Next.js 内置的 SEO 功能优化您的应用以适应搜索引擎。

## 元数据 API

Next.js 14+ 提供了强大的 SEO 元数据 API。

### 静态元数据

```tsx
// app/[locale]/about/page.tsx
import type { Metadata } from 'next';

export const metadata: Metadata = {
  title: '关于我们',
  description: '了解更多关于我们的公司和使命',
  keywords: ['关于', '公司', '使命'],
  authors: [{ name: '您的姓名' }],
  openGraph: {
    title: '关于我们',
    description: '了解更多关于我们的公司和使命',
    type: 'website',
    images: ['/og-image.jpg'],
  },
  twitter: {
    card: 'summary_large_image',
    title: '关于我们',
    description: '了解更多关于我们的公司和使命',
    images: ['/og-image.jpg'],
  },
};

export default function AboutPage() {
  return <div>关于我们</div>;
}
```

### 动态元数据

```tsx
// app/[locale]/blog/[slug]/page.tsx
import type { Metadata } from 'next';

type Props = {
  params: { slug: string };
};

export async function generateMetadata({ params }: Props): Promise<Metadata> {
  const post = await getPost(params.slug);

  return {
    title: post.title,
    description: post.excerpt,
    openGraph: {
      title: post.title,
      description: post.excerpt,
      type: 'article',
      publishedTime: post.publishedAt,
      authors: [post.author.name],
      images: [post.coverImage],
    },
  };
}

export default async function BlogPost({ params }: Props) {
  const post = await getPost(params.slug);
  return <article>{/* 内容 */}</article>;
}
```

## 开放图像

创建动态开放图像：

```tsx
// app/api/og/route.tsx
import { ImageResponse } from 'next/og';

export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);
  const title = searchParams.get('title') || '默认标题';

  return new ImageResponse(
    (
      <div
        style={{
          fontSize: 60,
          background: 'linear-gradient(to bottom, #4F46E5, #7C3AED)',
          width: '100%',
          height: '100%',
          display: 'flex',
          alignItems: 'center',
          justifyContent: 'center',
          color: 'white',
          fontWeight: 'bold',
        }}
      >
        {title}
      </div>
    ),
    {
      width: 1200,
      height: 630,
    }
  );
}
```

## 站点地图

自动生成站点地图：

```typescript
// app/sitemap.ts
import { MetadataRoute } from 'next';

export default function sitemap(): MetadataRoute.Sitemap {
  return [
    {
      url: 'https://yoursite.com',
      lastModified: new Date(),
      changeFrequency: 'yearly',
      priority: 1,
    },
    {
      url: 'https://yoursite.com/about',
      lastModified: new Date(),
      changeFrequency: 'monthly',
      priority: 0.8,
    },
    {
      url: 'https://yoursite.com/blog',
      lastModified: new Date(),
      changeFrequency: 'weekly',
      priority: 0.5,
    },
  ];
}
```

## Robots.txt

配置搜索引擎爬网：

```typescript
// app/robots.ts
import { MetadataRoute } from 'next';

export default function robots(): MetadataRoute.Robots {
  return {
    rules: {
      userAgent: '*',
      allow: '/',
      disallow: '/admin/',
    },
    sitemap: 'https://yoursite.com/sitemap.xml',
  };
}
```

## 结构化数据

添加 JSON-LD 结构化数据：

```tsx
export default function ProductPage({ product }) {
  const jsonLd = {
    '@context': 'https://schema.org',
    '@type': 'Product',
    name: product.name,
    image: product.image,
    description: product.description,
    offers: {
      '@type': 'Offer',
      price: product.price,
      priceCurrency: 'USD',
    },
  };

  return (
    <>
      <script
        type="application/ld+json"
        dangerouslySetInnerHTML={{ __html: JSON.stringify(jsonLd) }}
      />
      <div>{/* 产品内容 */}</div>
    </>
  );
}
```

## 规范 URL

设定规范 URL 以避免重复内容：

```tsx
export const metadata: Metadata = {
  alternates: {
    canonical: 'https://yoursite.com/page',
  },
};
```

## 最佳实践

1. **使用描述性标题**: 50-60 个字符
2. **撰写引人入胜的描述**: 150-160 个字符
3. **包含相关关键词**: 但避免堆砌关键词
4. **为图片添加替代文本**: 以更好地支持可访问性和 SEO
5. **使用语义 HTML**: 正确的标题层次结构 (h1, h2, h3)
6. **实施模式标记**: 帮助搜索引擎理解内容
7. **优化图片**: 使用 WebP 格式和合适的大小
8. **创建站点地图**: 帮助搜索引擎发现页面
